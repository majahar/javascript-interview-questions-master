function reverseString(str){
    const revStr = str.split("").reverse().join("");
    return revStr;
}
// Sol 2
function reverseString(str){
    let strlen = str.split("")
    let newStr='';
    for(let i=strlen.length; i>= 0; i--){
        if(strlen[i]){
            newStr += strlen[i];
        }
    }
    return newStr;

}
