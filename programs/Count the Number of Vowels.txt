//Count the Number of Vowels Using Regex

function countVowels(str){
    let vowels = ['a','e','i','o','u'];
    let count=0;
    str.split("").map((val,index) => {
        //console.log(val.toLowerCase())
        if(vowels.includes(val.toLowerCase())){
          console.log(val+"-->at index-> "+index+" Cont num:"+count++)
        }
        
    })
    
    return count;
}
