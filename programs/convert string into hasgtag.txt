//convert string into hasgtag


function hashTag(){
    //desired out put #MyNameIs MajaharKhanAndIAmJavascriptDeveloper
    let str= "my name is majahAr Khan and i am javascript developer";
    
    const hashTagStr= str.split(" ").map((val) => {
        
        // console.log(val.charAt(0).toUpperCase() +""+val.slice(1).toLowerCase())
        return val.charAt(0).toUpperCase()+val.slice(1).toLowerCase();
    }).join("")
return `#${hashTagStr}`;
    
}
